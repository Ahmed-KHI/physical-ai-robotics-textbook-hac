## ğŸ¨ Creating Environments

### Basic Scene Setup

```python
from omni.isaac.kit import SimulationApp

# Launch
simulation_app = SimulationApp({"headless": False})

import omni
from pxr import Usd, UsdGeom, Gf

# Get stage
stage = omni.usd.get_context().get_stage()

# Add ground plane
plane = UsdGeom.Mesh.Define(stage, "/World/Ground")
plane.CreatePointsAttr([(-10,-10,0), (10,-10,0), (10,10,0), (-10,10,0)])

# Add cube
cube = UsdGeom.Cube.Define(stage, "/World/Cube")
cube.AddTranslateOp().Set(Gf.Vec3d(0, 0, 1))
cube.CreateSizeAttr(1.0)
```

### Import Robot from URDF

```python
from omni.isaac.core.utils.extensions import enable_extension
enable_extension("omni.importer.urdf")

from omni.importer.urdf import _urdf

# Import URDF
_urdf.acquire_urdf_interface().import_urdf(
    urdf_path="/path/to/robot.urdf",
    dest_path="/World/Robot"
)
```

## ğŸ“· Synthetic Data Generation

### Setup Replicator

```python
import omni.replicator.core as rep

# Camera
camera = rep.create.camera(position=(5, 5, 5), look_at=(0, 0, 0))

# Render product
render_product = rep.create.render_product(camera, (1024, 1024))

# Writers
writer = rep.WriterRegistry.get("BasicWriter")
writer.initialize(output_dir="output", rgb=True, bounding_box_2d_tight=True)
writer.attach([render_product])

# Randomize and capture
with rep.trigger.on_frame(num_frames=100):
    with rep.create.group([cube]):
        rep.modify.pose(
            position=rep.distribution.uniform((-2,-2,0.5), (2,2,1.5))
        )
```

## ğŸ¤– Robot Control

### Articulation API

```python
from omni.isaac.core import World
from omni.isaac.core.robots import Robot

world = World()
robot = world.scene.add(Robot(prim_path="/World/Robot"))

# Set joint positions
joint_positions = [0.0, 1.57, -1.57, 0.0]
robot.set_joint_positions(joint_positions)

# Apply joint efforts (torques)
robot.apply_action(joint_efforts=[1.0, 0.5, 0.5, 0.0])
```

```python
from omni.isaac.core.prims import RigidPrim

# Create rigid body

rigid_cube.set_density(100.0)

# Set physics properties

```
